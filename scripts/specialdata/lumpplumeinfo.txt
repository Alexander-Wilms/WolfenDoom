/* Parser for PLUMEINF lumps, which define the actor class to spawn on a certain terrain texture.
 * Different than splashes because they are bigger, and spawned by GrenadeBase.SpawnGroundPlume.
 * The format is:
 * <Texture name> <Plume actor class name>
**/
class PlumeInfoLump {
	int lumpId;

	static PlumeInfoLump FromLump(String lumpName = "PLUMEINF") {
		PlumeInfoLump plumeInfo = new("PlumeInfoLump");
		plumeInfo.lumpId = Wads.FindLump(lumpName);
		return plumeInfo;
	}

	Dictionary Get() {
		Dictionary plumes = Dictionary.Create();
		return plumes;
	}
}
