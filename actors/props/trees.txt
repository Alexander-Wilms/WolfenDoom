ACTOR TreeBase : TreesBase
{
	+FLATSPRITE
	+ROLLSPRITE
	Radius 16
	Height 128
	States
	{
	Spawn:
		MDLA A 0 NODELAY A_JumpIf(CallACS("boa_bidiswitch")==1,"BidiSpawn")
	TridiSpawn:
		MDLA A -1
		Stop
	BidiSpawn:
		UNKN A -1 A_SetPitch(270+pitch) //fix the pitch, also spawn the UNKN sprite if not set, thanks to inheritances
		Stop
	}
}

ACTOR BigBush : SmallBush1
{
	//$Title Bush (big, rippable)
	Height 32
	Health 1
	+SOLID
	+DONTTHRUST
	+NOBLOOD
	+NOTAUTOAIMED
	+SHOOTABLE
	DistanceCheck "boa_treeslod"
	States
	{
	Spawn:
		BUSZ A -1 NODELAY A_SetScale(ScaleX * RandomPick(-1.0, 1.0), ScaleY)
		Stop
	Death:
		"####" "#" 0 A_UnSetSolid
		"####" "#" 0 A_StartSound("GRASBRKS", CHAN_AUTO, 0, frandom (0.5,0.8), ATTN_NORM)
		"####" "####" 0 A_SpawnItemEx("Debris_Leaf", random(0,4), random(0,8), random(0,24), random(1,3), random(1,3), random(1,3), random(0,360), SXF_CLIENTSIDE)
		"####" "#" 2 A_SpawnItemEx("GrassFrags")
		BUSZ B -1
		Stop
	}
}

ACTOR RandomRye : FlattenableProp
{
	//$Category Flora (BoA)
	//$Title Random Rye
	//$Color 3
	//$Sprite RYE_A0
	Radius 8
	Height 32
	Scale 0.8
	DistanceCheck "boa_grasslod"
	States
	{
	Spawn:
		RYE_ A 0 NODELAY A_Jump(256, "RyeA", "RyeB", "RyeC", "RyeD", "RyeE", "RyeF")
		Stop
	RyeA:
		RYE_ A 0
		Goto Setup
	RyeB:
		RYE_ B 0
		Goto Setup
	RyeC:
		RYE_ C 0
		Goto Setup
	RyeD:
		RYE_ D 0
		Goto Setup
	RyeE:
		RYE_ E 0
		Goto Setup
	RyeF:
		RYE_ F 0
		Goto Setup
	Setup:
		"####" "#" 0 A_SetScale(ScaleX * frandom(0.85, 1.1), ScaleY * frandom(0.85, 1.1))
		"####" "#" -1 A_SetScale(ScaleX * RandomPick(-1.0, 1.0), ScaleY)
		Stop
	}
}
