ACTOR SkyboxTracerTrail //3d actor
{
	Alpha 1.0
	Scale 1.0
	RenderStyle Add
	+BRIGHT
	+CLIENTSIDEONLY
	+NOINTERACTION
	States
	{
	Spawn:
		MDLA A 70
		Goto Death
	Death:
		MDLA A 4 A_FadeOut(0.1)
		Loop
	}
}

ACTOR SkyboxTracerSpawner : SwitchableDecoration
{
	//$Category Special Effects (BoA)
	//$Title Skyfire Tracers
	//$Color 12
	//$Sprite UNKNA0
	+INVULNERABLE
	+NOGRAVITY
	+NOINTERACTION
	+SHOOTABLE
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_JumpIf(CallACS("boa_smokeswitch")==0,"EndSpawn")
	Active:
		TNT1 A 12 A_SpawnItemEx("SkyboxTracerTrail", 0, 0, 0, 1, 0, 8, 0, SXF_TRANSFERPITCH | SXF_TRANSFERROLL | SXF_TRANSFERSCALE)
		Loop
	Inactive:
		TNT1 A -1
		Stop
	EndSpawn:
		TNT1 A 1
		Stop
	}
}
