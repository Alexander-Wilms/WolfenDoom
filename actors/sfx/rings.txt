ACTOR BaseOrb : ParticleBase
{
	Radius 0
	Height 0
	-SOLID
	+BRIGHT
	+MISSILE
	+NOBLOCKMAP
	+NOGRAVITY
	+NOINTERACTION
	RenderStyle "Add"
	Alpha 0.01
	YScale 0.1
	XScale 0.4
	States
	{
	Spawn:
		LRFX AAAAA 1 A_FadeIn(0.2)
		"####" AAAAAAAAAAAAAAAA 1 A_FadeOut(0.07)
		Stop
	}
}
