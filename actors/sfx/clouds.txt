//Base Cloud Grey
ACTOR CloudBaseGrey : ParticleBase
{
	+MISSILE
	+NOBLOCKMAP
	+NOGRAVITY
	Radius 1
	Height 1
	RenderStyle "Translucent"
	Alpha 0.1
	Scale 0.5
	+ParticleBase.CHECKPOSITION
	States
	{
	Spawn:
		TNT1 A 0 NODELAY A_Jump(256, "Cloud1", "Cloud2")
	Cloud1:
		CLXG A 0
		Goto Movement
	Cloud2:
		CLXG B 0
		Goto Movement
	Movement:
		"####""###########" 3 A_FadeIn(0.05)
		"####" "#" 1000
		Wait
	}
}

ACTOR SmallCloudGrey : CloudBaseGrey  {	Scale 1.0 }
ACTOR MediumCloudGrey : CloudBaseGrey {	Scale 2.0 }
ACTOR LargeCloudGrey : CloudBaseGrey  {	Scale 4.0 }
