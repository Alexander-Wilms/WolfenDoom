ACTOR SingleCoin : CoinItem
{
	//$Category Pickups (BoA)/Treasures
	//$Title Treasure (Coin, 1pts)
	//$Color 17
	Scale 0.5
	Inventory.Amount 1
	Inventory.PickupMessage "$COIN"
	Inventory.PickupSound "treasure/pickup"
	States
	{
	Spawn:
		ONEC A -1 LIGHT(COINLITE)
		Stop
	}
}
